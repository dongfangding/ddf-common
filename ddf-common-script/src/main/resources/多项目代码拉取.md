
多模块项目， 聚合到一个统一的文件目录内

### 切换分支

```shell script
# bin/bash

branch=$1
root=/d/sian-modules
source_root=/d/sian/
dir=$(ls ${source_root})

for curr in ${dir}
do
  cd ${source_root}"${curr}" || exit
  echo "开始检索[${source_root}${curr}]的[${branch}]分支代码"
  git pull checkout "${branch}"
done;

cd ${root} || exit
```

### merge

```shell script
# bin/bash

project=$1
fromBranch=$2
targetBranch=$3
source_root=/d/sian/

if [ ! "${project}" ] ; then
  echo "请输入要merge的项目"
  exit
fi;

if [ ! "${fromBranch}" ] ; then
  echo "请输入要merge的from分支"
  exit
fi;

if [ ! "${targetBranch}" ] ; then
  echo "请输入要merge的to分支"
  exit
fi;

cd ${source_root}"${project}" || exit

git checkout "${targetBranch}"
git pull origin "${targetBranch}"
git merge "${fromBranch}"
git push origin "${targetBranch}"
git checkout "${fromBranch}"
git pull origin "${fromBranch}"

```
